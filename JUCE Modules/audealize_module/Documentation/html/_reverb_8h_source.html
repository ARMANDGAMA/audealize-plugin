44  void processMonoBlock (float* channelData, int blockSize)
45  {
46  float samp, sampRev, sampDry, sampOut;
47 
48  for (int i = 0; i < blockSize; i++)
49  {
50  sampDry = channelData[i];
51 
52  // Process sample through comb filter network
53  sampRev = processCombs (sampDry * wet);
54 
55  // Process allpass filter
56  sampRev = mAllpass[0].process_allpass_comb (sampRev, mDelayVal[0] * mSampleRate, ALLPASSGAIN);
57 
58  // Process lowpass filter
59  sampRev = mLowpass.processSample (sampRev, 0);
60 
61  sampRev *= gain;
62 
63  // Delay unprocessed signal to match phase shift caused by the delayed comb filters
64  samp = wet * mDelay[0].process (sampDry, MINDELAY * mSampleRate);
65 
66  samp *= gainclean;
67 
68  // Average clean and filtered samples
69  samp = (samp + sampRev) * .5f;
70 
71  samp *= gainscale;
72 
73  sampDry *= dry;
74 
75  // Write processed sample back to the buffer
76  sampOut = samp + sampDry;
77  // DBG ("processing mono");
78 
79  JUCE_UNDENORMALISE (sampOut);
80 
81  channelData[i] = sampOut;
82  }
