551 inline double convert_periodic (double val, periodic_unit unit_in, periodic_unit unit_out)
552 {
553  // calculates different periodical units into each other.
554  // it is used for e.g. pulsator or vintage delay to unify
555  // the selected timing method
556  if (unit_in == unit_out)
557  return val;
558  double freq = 0.;
559  switch (unit_in) {
560  case UNIT_BPM: freq = val / 60.; break;
561  case UNIT_MS: freq = 1. / (val / 1000.); break;
562  case UNIT_HZ: freq = val; break;
563  case UNIT_SYNC: freq = val / 60.; break;
564  }
565  switch (unit_out) {
566  default:
567  case UNIT_BPM: return freq * 60.;
568  case UNIT_MS: return (1. / freq) / 1000.;
569  case UNIT_HZ: return freq;
570  case UNIT_SYNC: return freq * 60.;
571  }
572 };
