48  void processMonoBlock (float* channelData, int blockSize)
49  {
50  float samp, sampRev, sampDry, sampOut;
51 
52  for (int i = 0; i < blockSize; i++)
53  {
54  sampDry = channelData[i];
55 
56  // Process sample through comb filter network
57  sampRev = processCombs (sampDry * wet);
58 
59  // Process allpass filter
60  sampRev = mAllpass[0].process_allpass_comb (sampRev, mDelayVal[0] * mSampleRate, ALLPASSGAIN);
61 
62  // Process lowpass filter
63  sampRev = mLowpass.processSample (sampRev, 0);
64 
65  sampRev *= gain;
66 
67  // Delay unprocessed signal to match phase shift caused by the delayed comb filters
68  samp = wet * mDelay[0].process (sampDry, MINDELAY * mSampleRate);
69 
70  samp *= gainclean;
71 
72  // Average clean and filtered samples
73  samp = (samp + sampRev) * .5f;
74 
75  samp *= gainscale;
76 
77  sampDry *= dry;
78 
79  // Write processed sample back to the buffer
80  sampOut = samp + sampDry;
81  // DBG ("processing mono");
82 
83  JUCE_UNDENORMALISE (sampOut);
84  channelData[i] = sampOut;
85  }
86  }

96  {
