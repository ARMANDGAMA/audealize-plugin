157 void AudealizereverbAudioProcessor::processBlock (AudioSampleBuffer& buffer, MidiBuffer& midiMessages)
158 {
159  const int totalNumInputChannels = getTotalNumInputChannels ();
160  const int totalNumOutputChannels = getTotalNumOutputChannels ();
161 
162  // In case we have more outputs than inputs, this code clears any output
163  // channels that didn't contain input data, (because these aren't
164  // guaranteed to be empty - they may contain garbage).
165  // This is here to avoid people getting screaming feedback
166  // when they first compile a plugin, but obviously you don't need to keep
167  // this code if your algorithm always overwrites all the output channels.
168  for (int i = totalNumInputChannels; i < totalNumOutputChannels; ++i) buffer.clear (i, 0, buffer.getNumSamples ());
169 
170  // Parameter smoothing
171  float diff;
172 
173  diff = fabs (mReverb.get_d () - mSmoothedVals[kParamD].getTargetValue ());
174  if (diff > 0.01f * mSmoothedVals[kParamD].getTargetValue ())
175  {
176  mReverb.set_d (mSmoothedVals[kParamD].getNextValue ());
177  }
178 
179  diff = fabs (mReverb.get_g () - mSmoothedVals[kParamG].getTargetValue ());
180  if (diff > 0.01f * mSmoothedVals[kParamG].getTargetValue ())
181  {
182  mReverb.set_g (mSmoothedVals[kParamG].getNextValue ());
183  }
184 
185  diff = fabs (mReverb.get_m () - mSmoothedVals[kParamM].getTargetValue ());
186  if (diff > 0.01f * mSmoothedVals[kParamM].getTargetValue ())
187  {
188  mReverb.set_m (mSmoothedVals[kParamM].getNextValue ());
189  }
190 
191  diff = fabs (mReverb.get_f () - mSmoothedVals[kParamF].getTargetValue ());
192  if (diff > 0.01f * mSmoothedVals[kParamF].getTargetValue ())
193  {
194  mReverb.set_f (mSmoothedVals[kParamF].getNextValue ());
195  }
196 
197  diff = fabs (mReverb.get_E () - mSmoothedVals[kParamE].getTargetValue ());
198  if (diff > 0.01f * mSmoothedVals[kParamE].getTargetValue ())
199  {
200  mReverb.set_E (mSmoothedVals[kParamE].getNextValue ());
201  }
202 
203  diff = fabs (mReverb.get_wetdry () - mSmoothedVals[kParamAmount].getTargetValue ());
204  if (diff > 0.01f * mSmoothedVals[kParamAmount].getTargetValue ())
205  {
206  mReverb.set_wetdry (mSmoothedVals[kParamAmount].getNextValue ());
207  }
208  // end parameter smoothing
209 
210  // Process reverb
211  if (!mBypass)
212  {
213  if (totalNumInputChannels == 1)
214  {
215  float* channelData = buffer.getWritePointer (0);
216 
217  mReverb.processMonoBlock (channelData, buffer.getNumSamples ());
218  }
219  else
220  {
221  float* channelData1 = buffer.getWritePointer (0);
222  float* channelData2 = buffer.getWritePointer (1);
223 
224  mReverb.processStereoBlock (channelData1, channelData2, buffer.getNumSamples ());
225  }
226  }
